\newcommand{\famphoto}[2]{%
  \begin{figure}[h]
    \centering
    \includegraphics[width=0.7\linewidth]{photos/#1}
    \caption{#2}
  \end{figure}%
}
\newcommand{\fullphoto}[2]{%
  \begin{figure}[p]
    \centering
    \includegraphics[width=\linewidth]{photos/#1}
    \caption{#2}
  \end{figure}%
}
\newcommand{\marginphoto}[2]{%
  \marginpar{%
    \includegraphics[width=\marginparwidth]{photos/#1}\\
    \footnotesize #2%
  }%
}
